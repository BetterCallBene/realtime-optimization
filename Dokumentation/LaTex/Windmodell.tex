\section{Modellierung Wind}

\subsection{Kraft:}

\textbf{Winddruck:} 
\[ P_{Wind} = c_P \cdot \frac{\rho_{Luft}}{2} \cdot v^2 \]

\textbf{Windkraft:}
\[ F_{Wind}= A \cdot P_{Wind} \]

\begin{tabular}{ccc}
	\(c_P\) & Druckbeiwert & z.B. \(c_P=0.7-2\) \\
	\(\rho_{Luft}\) & Luftdichte in \(\frac{kg}{m^3}\) & abhängig von Höhe und Temperatur (ca.\(1.0 - 1.3 \frac{kg}{m^3})\)
\end{tabular}
